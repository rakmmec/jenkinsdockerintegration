pipeline{

	agent any

	environment {
		DOCKERHUB_CREDENTIALS=credentials('docker-hub-credential')
	}

	stages {

		stage('Build') {

			steps {
				sh 'docker build -t rak161186/testingapp:latest .'
			}
		}

		stage('Login') {

			steps {
				sh 'echo $DOCKERHUB_CREDENTIALS_PSW | docker login -u $DOCKERHUB_CREDENTIALS_USR --password-stdin'
			}
		}

		stage('Push') {

			steps {
				sh 'docker push rak161186/testingapp:latest'
			}
		}
	}

	post {
		always {
			sh 'docker logout'
		}
	}

}
