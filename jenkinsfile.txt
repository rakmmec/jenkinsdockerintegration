pipeline{

	agent any

	environment {
		DOCKERHUB_CREDENTIALS=credentials('docker-hub-credential')
	}

	stages {

		stage('Build') {

			steps {
				sh 'docker build -t rak161186/testingapp:latest .'
			}
		}

		
		stage('Push image') {
			withDockerRegistry([ credentialsId: "docker-hub-credentials", url: "" ]) {
			bat "docker push rak161186/teamcitydocker:build"
			}
			}
	}
}
